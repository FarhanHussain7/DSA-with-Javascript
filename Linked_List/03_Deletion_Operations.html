deleteFromTail() { // Check if list is empty if (!this.head) { return null; } // If only one node exists if (!this.head.next) { const deletedData = this.head.data; this.head = null; this.size--; return deletedData; } // Traverse to second-to-last node let current = this.head; while (current.next && current.next.next) { current = current.next; } // Store data to return const deletedData = current.next.data; // Remove last node current.next = null; // Decrement size this.size--; return deletedData; }

deleteAt(index) { // Check for valid index if (index < 0 || index >= this.size) { return null; // Invalid index } // If deleting head, use deleteFromHead if (index === 0) { return this.deleteFromHead(); } // Traverse to node before deletion point let current = this.head; for (let i = 0; i < index - 1; i++) { current = current.next; } // Store data to return const deletedData = current.next.data; // Skip the node to delete current.next = current.next.next; // Decrement size this.size--; return deletedData; }

deleteValue(value) { // Check if list is empty if (!this.head) { return false; } // If head contains the value if (this.head.data === value) { this.head = this.head.next; this.size--; return true; } // Traverse to find the value let current = this.head; while (current.next && current.next.data !== value) { current = current.next; } // If value found, delete it if (current.next) { current.next = current.next.next; this.size--; return true; } return false; // Value not found } // Delete all occurrences of a value deleteAll(value) { let deleted = false; // Delete from head while it matches while (this.head && this.head.data === value) { this.head = this.head.next; this.size--; deleted = true; } // Delete from middle and end let current = this.head; while (current && current.next) { if (current.next.data === value) { current.next = current.next.next; this.size--; deleted = true; } else { current = current.next; } } return deleted; }
