âœ… Object.keys(), Object.values(), Object.entries()
Extract keys, values, or key-value pairs from an object as arrays. 
const user = {
    name: "John",
    age: 30,
    city: "New York",
    email: "john@example.com"
};

// Get all keys
const keys = Object.keys(user);
console.log(keys);
// ["name", "age", "city", "email"]

// Get all values
const values = Object.values(user);
console.log(values);
// ["John", 30, "New York", "john@example.com"]

// Get key-value pairs
const entries = Object.entries(user);
console.log(entries);
// [["name", "John"], ["age", 30], ["city", "New York"], ["email", "john@example.com"]]

// Practical usage
entries.forEach(([key, value]) => {
    console.log(key + ": " + value);
});
âœ… Object.freeze() - Prevent All Modifications
Make an object immutable - cannot add, delete, or modify properties. 
const user = {
    name: "John",
    age: 30,
    email: "john@example.com"
};

// Freeze the object
const frozenUser = Object.freeze(user);

console.log(Object.isFrozen(frozenUser)); // true

// Try to modify
frozenUser.name = "Bob"; // Fails silently
console.log(frozenUser.name); // "John" (unchanged)

// Try to add new property
frozenUser.phone = "555-1234"; // Fails silently
console.log(frozenUser.phone); // undefined

// Try to delete property
delete frozenUser.age; // Fails silently
console.log(frozenUser.age); // 30 (still exists)

// In strict mode: throws error
"use strict";
frozenUser.city = "NYC"; // TypeError: Cannot assign to read only property

âœ… Object.assign() - Copy Properties
Copy all enumerable properties from one or more source objects to a target object. 
// Syntax: Object.assign(target, source1, source2, ...)

const defaults = {
    theme: "light",
    language: "en",
    notifications: true,
    fontSize: 14
};

const userPreferences = {
    theme: "dark",
    notifications: false
};

// Merge into new object
const settings = Object.assign({}, defaults, userPreferences);
console.log(settings);
// { theme: "dark", language: "en", notifications: false, fontSize: 14 }

// Copy into existing object (modifies it)
const config = Object.assign(defaults, userPreferences);
console.log(config === defaults); // true (same object)

// Multiple sources
const obj1 = { a: 1 };
const obj2 = { b: 2 };
const obj3 = { c: 3 };
const merged = Object.assign({}, obj1, obj2, obj3);
console.log(merged); // { a: 1, b: 2, c: 3 }
âœ… Object.getOwnPropertyNames() & getOwnPropertyDescriptor()
Get property names and their descriptors (writable, enumerable, configurable). 
const obj = {};

Object.defineProperty(obj, "name", {
    value: "John",
    writable: true,
    enumerable: true,
    configurable: true
});

Object.defineProperty(obj, "secret", {
    value: "hidden",
    writable: false,
    enumerable: false,
    configurable: false
});

// Get all own property names
const names = Object.getOwnPropertyNames(obj);
console.log(names); // ["name", "secret"]

// Get descriptor for single property
const nameDescriptor = Object.getOwnPropertyDescriptor(obj, "name");
console.log(nameDescriptor);
// { value: "John", writable: true, enumerable: true, configurable: true }

// Get all descriptors
const descriptors = Object.getOwnPropertyDescriptors(obj);
console.log(descriptors);
// Shows all properties with their descriptors

// Object.keys() vs getOwnPropertyNames()
console.log(Object.keys(obj)); // ["name"] (enumerable only)
console.log(Object.getOwnPropertyNames(obj)); // ["name", "secret"] (all)

âœ… Object.getPrototypeOf() & Object.setPrototypeOf()
Get or set the prototype of an object for inheritance. 
// Create prototype object
const animalProto = {
    speak() { return this.name + " makes a sound"; }
};

// Create object with prototype
const dog = Object.create(animalProto);
dog.name = "Buddy";

// Get prototype
console.log(Object.getPrototypeOf(dog) === animalProto); // true

// Use inherited method
console.log(dog.speak()); // "Buddy makes a sound"

// Create new prototype
const dogProto = {
    speak() { return this.name + " barks"; },
    wagTail() { return this.name + " wags tail"; }
};

// Set new prototype
Object.setPrototypeOf(dog, dogProto);
console.log(Object.getPrototypeOf(dog) === dogProto); // true
console.log(dog.speak()); // "Buddy barks"
console.log(dog.wagTail()); // "Buddy wags tail"
âœ… Object.fromEntries() - Convert Entries to Object
Create an object from key-value pairs (reverse of Object.entries()). 
// From array of entries
const entries = [
    ["name", "John"],
    ["age", 30],
    ["city", "New York"]
];

const obj = Object.fromEntries(entries);
console.log(obj);
// { name: "John", age: 30, city: "New York" }

// From Map
const map = new Map([
    ["firstName", "John"],
    ["lastName", "Doe"]
]);

const user = Object.fromEntries(map);
console.log(user);
// { firstName: "John", lastName: "Doe" }

// Converting Object to Map and back
const original = { x: 10, y: 20 };
const m = new Map(Object.entries(original));
const restored = Object.fromEntries(m);
console.log(original);
console.log(restored);
console.log(JSON.stringify(original) === JSON.stringify(restored)); // true
