âœ… Object Literal Notation
The simplest way to create an object using curly braces and key-value pairs. 
// Syntax
const objectName = {
    key1: value1,
    key2: value2,
    key3: value3
};

// Example
const person = {
    name: "John",
    age: 30,
    city: "New York",
    email: "john@example.com"
};

console.log(person);
console.log(person.name); // "John"
âœ… Constructor Function
Create reusable object templates using a function called with the 'new' keyword. 
// Define constructor function
function Person(name, age, city) {
    this.name = name;
    this.age = age;
    this.city = city;
    this.greet = function() {
        return "Hello, I'm " + this.name;
    };
}

// Create instances
const person1 = new Person("John", 30, "NYC");
const person2 = new Person("Alice", 25, "LA");

console.log(person1);
console.log(person1.greet());
console.log(person2);
âœ… Object.create() Method
Create an object with a specific prototype. Useful for inheritance patterns. 
// Create prototype object
const PersonProto = {
    greet() {
        return "Hello, I'm " + this.name;
    },
    introduce() {
        return this.name + " is " + this.age + " years old";
    }
};

// Create new object with PersonProto as prototype
const person1 = Object.create(PersonProto);
person1.name = "John";
person1.age = 30;

const person2 = Object.create(PersonProto);
person2.name = "Alice";
person2.age = 25;

console.log(person1.name); // "John"
console.log(person1.greet()); // "Hello, I'm John"
console.log(person2.introduce()); // "Alice is 25 years old"
âœ… Object.assign() Method
Copy properties from one or more source objects to a target object. 
// Syntax: Object.assign(target, source1, source2, ...)

const defaultSettings = {
    theme: "light",
    language: "en",
    notifications: true
};

const userSettings = {
    theme: "dark",
    notifications: false
};

// Merge objects
const finalSettings = Object.assign({}, defaultSettings, userSettings);

console.log(finalSettings);
// Output: { theme: "dark", language: "en", notifications: false }

// Direct merge (modifies target)
const config = Object.assign(defaultSettings, userSettings);
console.log(config);
console.log(config === defaultSettings); // true (same object)
âœ… ES6 Class Syntax
Modern syntax for creating objects with constructor and methods. 
// Class definition
class Car {
    constructor(brand, model, year) {
        this.brand = brand;
        this.model = model;
        this.year = year;
    }
    
    // Instance method
    getInfo() {
        return this.year + " " + this.brand + " " + this.model;
    }
    
    // Static method
    static compare(car1, car2) {
        return car1.year - car2.year;
    }
}

// Create instances
const car1 = new Car("Toyota", "Camry", 2020);
const car2 = new Car("Honda", "Civic", 2022);

console.log(car1.getInfo()); // "2020 Toyota Camry"
console.log(car2.getInfo()); // "2022 Honda Civic"
console.log(Car.compare(car1, car2)); // -2
âœ… Singleton Pattern
Create only one instance of an object and reuse it throughout the application. 
// Singleton implementation
const DatabaseConnection = (() => {
    let instance;
    
    return {
        getInstance() {
            if (!instance) {
                instance = {
                    connectionId: Math.random(),
                    connect() { return "Connected"; },
                    disconnect() { return "Disconnected"; }
                };
            }
            return instance;
        }
    };
})();

const db1 = DatabaseConnection.getInstance();
const db2 = DatabaseConnection.getInstance();

console.log(db1 === db2); // true (same instance)
console.log(db1.connectionId === db2.connectionId); // true
âœ… Factory Function
Use a function to create and configure objects based on parameters. 
// Factory function
function createUser(name, role) {
    return {
        name: name,
        role: role,
        permissions: getPermissions(role),
        getInfo() {
            return this.name + " is a " + this.role;
        }
    };
}

function getPermissions(role) {
    switch(role) {
        case "admin": return ["read", "write", "delete"];
        case "user": return ["read", "write"];
        case "guest": return ["read"];
        default: return [];
    }
}

const admin = createUser("John", "admin");
const user = createUser("Alice", "user");
const guest = createUser("Bob", "guest");

console.log(admin);
console.log(admin.getInfo());
console.log(user.permissions);
