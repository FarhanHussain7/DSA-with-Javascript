Access, Add, Modify, and Delete Object Properties

âœ… Dot Notation vs Bracket Notation
Two different ways to access object properties. Bracket notation is more flexible. 
const person = {
    name: "John",
    age: 30,
    "email-address": "john@email.com",
    "phone number": "+1-555-1234"
};

// Dot Notation (simple property names only)
console.log(person.name); // "John"
console.log(person.age); // 30

// Bracket Notation (works for all property names)
console.log(person["name"]); // "John"
console.log(person["age"]); // 30
console.log(person["email-address"]); // "john@email.com"
console.log(person["phone number"]); // "+1-555-1234"

// Dynamic access with variable
const key = "age";
console.log(person[key]); // 30

// Using expressions in bracket notation
const index = "na" + "me";
console.log(person[index]); // "John"
âœ… Adding New Properties
Multiple ways to add new properties to an existing object. 
const obj = { name: "John" };
console.log(obj); // { name: "John" }

// Method 1: Dot notation
obj.age = 30;
console.log(obj); // { name: "John", age: 30 }

// Method 2: Bracket notation
obj["email"] = "john@email.com";
console.log(obj); // { name: "John", age: 30, email: "john@email.com" }

// Method 3: Object.assign()
Object.assign(obj, { city: "NYC", country: "USA" });
console.log(obj);
// { name: "John", age: 30, email: "john@email.com", city: "NYC", country: "USA" }

// Method 4: Spread operator
const newObj = { ...obj, phone: "555-1234" };
console.log(newObj);
âœ… Modifying Existing Properties
Update existing property values in different ways. 
const user = {
    name: "Alice",
    age: 25,
    email: "alice@email.com"
};

// Method 1: Dot notation
user.name = "Bob";
console.log(user.name); // "Bob"

// Method 2: Bracket notation
user["age"] = 30;
console.log(user["age"]); // 30

// Method 3: Using Object.defineProperty()
Object.defineProperty(user, "phone", {
    value: "555-1234",
    writable: true,
    enumerable: true,
    configurable: true
});

// Method 4: Update with Object.assign()
Object.assign(user, { email: "bob@email.com", city: "LA" });

console.log(user);
// { name: "Bob", age: 30, email: "bob@email.com", phone: "555-1234", city: "LA" }
âœ… Deleting Properties
Remove properties from objects using the delete operator. 
const obj = {
    name: "John",
    age: 30,
    email: "john@email.com",
    phone: "555-1234"
};

console.log(obj);
// { name: "John", age: 30, email: "john@email.com", phone: "555-1234" }

// Delete property
delete obj.email;
console.log(obj);
// { name: "John", age: 30, phone: "555-1234" }

console.log("email" in obj); // false
console.log(obj.email); // undefined

// Delete multiple properties
delete obj.phone;
delete obj.age;
console.log(obj);
// { name: "John" }

// Check existence
console.log(Object.hasOwn(obj, "name")); // true
console.log(Object.hasOwn(obj, "email")); // false
âœ… Using Object.defineProperty()
Define properties with specific descriptors (writable, enumerable, configurable). 
const obj = {};

// Define a read-only property
Object.defineProperty(obj, "name", {
    value: "John",
    writable: false,    // Cannot be modified
    enumerable: true,   // Shows in Object.keys()
    configurable: true  // Can be deleted/reconfigured
});

// Define multiple properties at once
Object.defineProperties(obj, {
    age: {
        value: 30,
        writable: true,
        enumerable: true
    },
    email: {
        value: "john@email.com",
        writable: true,
        enumerable: false  // Won't show in Object.keys()
    }
});

console.log(obj.name); // "John"
obj.name = "Bob"; // Fails silently (writable: false)
console.log(obj.name); // "John" (unchanged)

console.log(Object.keys(obj)); // ["name", "age"] (email excluded)
console.log(obj.email); // "john@email.com" (still accessible)
âœ… Dynamic Property Access
Access and modify properties using variables and computed values. 
const obj = {
    firstName: "John",
    lastName: "Doe",
    email: "john@email.com"
};

// Using variable as key
let key = "firstName";
console.log(obj[key]); // "John"

// Changing the key
key = "email";
console.log(obj[key]); // "john@email.com"

// Using computed property names
const prop = "last" + "Name";
console.log(obj[prop]); // "Doe"

// Using loop with dynamic access
const keys = Object.keys(obj);
for (let i = 0; i < keys.length; i++) {
    console.log(keys[i] + ": " + obj[keys[i]]);
}

// Function to get property
function getProperty(obj, propertyName) {
    return obj[propertyName];
}

console.log(getProperty(obj, "firstName")); // "John"
âœ… Checking Property Existence
Verify if a property exists in an object before accessing it. 
const obj = { name: "John", age: 30 };

// Method 1: 'in' operator
console.log("name" in obj); // true
console.log("email" in obj); // false

// Method 2: hasOwnProperty()
console.log(obj.hasOwnProperty("name")); // true
console.log(obj.hasOwnProperty("age")); // true

// Method 3: Object.hasOwn() (recommended)
console.log(Object.hasOwn(obj, "name")); // true
console.log(Object.hasOwn(obj, "email")); // false

// Method 4: Undefined check
console.log(obj.name !== undefined); // true
console.log(obj.email !== undefined); // false

// Difference: 'in' includes inherited properties
const proto = { city: "NYC" };
const user = Object.create(proto);
user.name = "John";

console.log("name" in user); // true
console.log("city" in user); // true (inherited)
console.log(user.hasOwnProperty("city")); // false
