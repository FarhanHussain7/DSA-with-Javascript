1. Object Creation Methods

‚úÖ Literal Notation
// Object Literal
const person = {
    name: "John",
    age: 30,
    city: "New York"
};
console.log(person); // { name: "John", age: 30, city: "New York" }
‚úÖ Constructor Function
// Constructor Function
function Person(name, age) {
    this.name = name;
    this.age = age;
}
const person1 = new Person("Alice", 25);
console.log(person1); // Person { name: "Alice", age: 25 }
‚úÖ Object.create()
// Using Object.create()
const proto = { greet() { return "Hello"; } };
const obj = Object.create(proto);
obj.name = "John";
console.log(obj.name); // "John"
console.log(obj.greet()); // "Hello"
‚úÖ Object.assign()
// Using Object.assign()
const defaults = { name: "", age: 0 };
const user = { name: "Bob", age: 30 };
const merged = Object.assign({}, defaults, user);
console.log(merged); // { name: "Bob", age: 30 }
‚úÖ ES6 Class Syntax
// Class Declaration
class Car {
    constructor(brand, color) {
        this.brand = brand;
        this.color = color;
    }
}
const car1 = new Car("Toyota", "Red");
console.log(car1); // Car { brand: "Toyota", color: "Red" }

2. Object Access & Modification Operations

‚úÖ Dot Notation vs Bracket Notation
const obj = { name: "John", age: 30, "email-id": "john@email.com" };

// Dot Notation
console.log(obj.name); // "John"

// Bracket Notation
console.log(obj["age"]); // 30
console.log(obj["email-id"]); // "john@email.com"

// Dynamic access
const key = "name";
console.log(obj[key]); // "John"
‚úÖ Adding Properties
const obj = { name: "John" };

// Add property using dot notation
obj.age = 30;

// Add property using bracket notation
obj["email"] = "john@email.com";

// Add multiple properties
Object.assign(obj, { city: "NYC", country: "USA" });

console.log(obj);
// { name: "John", age: 30, email: "john@email.com", city: "NYC", country: "USA" }
‚úÖ Modifying Properties
const user = { name: "Alice", age: 25 };

// Modify existing properties
user.name = "Bob";
user["age"] = 30;

// Using Object.defineProperty()
Object.defineProperty(user, "email", {
    value: "bob@email.com",
    writable: true,
    enumerable: true
});

console.log(user);
// { name: "Bob", age: 30, email: "bob@email.com" }
‚úÖ Deleting Properties
const obj = { name: "John", age: 30, email: "john@email.com" };

// Delete property
delete obj.email;

console.log(obj); // { name: "John", age: 30 }
console.log("email" in obj); // false
console.log(obj.email); // undefined

3. Object Static Methods

‚úÖ Object.keys(), Object.values(), Object.entries()
const user = { name: "John", age: 30, city: "NYC" };

// Get all keys
console.log(Object.keys(user));
// ["name", "age", "city"]

// Get all values
console.log(Object.values(user));
// ["John", 30, "NYC"]

// Get key-value pairs
console.log(Object.entries(user));
// [["name", "John"], ["age", 30], ["city", "NYC"]]
‚úÖ Object.getOwnPropertyNames() & Object.getOwnPropertySymbols()
const sym = Symbol("id");
const obj = { name: "John", age: 30, [sym]: 123 };

// Get property names (strings and symbols excluded)
console.log(Object.getOwnPropertyNames(obj));
// ["name", "age"]

// Get symbol properties
console.log(Object.getOwnPropertySymbols(obj));
// [Symbol(id)]
‚úÖ Object.freeze(), Object.seal(), Object.preventExtensions()
// Object.freeze() - Cannot add, delete, or modify
const frozen = Object.freeze({ name: "John" });
frozen.age = 30; // Silently fails (strict mode: Error)

// Object.seal() - Cannot add or delete, but can modify
const sealed = Object.seal({ name: "John" });
sealed.name = "Bob"; // ‚úì Works
sealed.age = 30; // Silently fails

// Object.preventExtensions() - Cannot add, but can delete/modify
const restricted = Object.preventExtensions({ name: "John" });
restricted.name = "Bob"; // ‚úì Works
delete restricted.name; // ‚úì Works
restricted.age = 30; // Silently fails
‚úÖ Object.isFrozen(), Object.isSealed(), Object.isExtensible()
const obj = {};
console.log(Object.isExtensible(obj)); // true

Object.seal(obj);
console.log(Object.isSealed(obj)); // true
console.log(Object.isFrozen(obj)); // false

const frozen = Object.freeze({});
console.log(Object.isFrozen(frozen)); // true
‚úÖ Object.getPrototypeOf() & Object.setPrototypeOf()
const proto = { greet() { return "Hello"; } };
const obj = Object.create(proto);

// Get prototype
console.log(Object.getPrototypeOf(obj) === proto); // true

// Set prototype
const newProto = { sayBye() { return "Bye"; } };
Object.setPrototypeOf(obj, newProto);
console.log(Object.getPrototypeOf(obj) === newProto); // true
‚úÖ Object.getOwnPropertyDescriptor() & Object.getOwnPropertyDescriptors()
const obj = { name: "John" };

// Get descriptor for single property
const descriptor = Object.getOwnPropertyDescriptor(obj, "name");
console.log(descriptor);
// { value: "John", writable: true, enumerable: true, configurable: true }

// Get descriptors for all properties
const descriptors = Object.getOwnPropertyDescriptors(obj);
console.log(descriptors);
// { name: { value: "John", writable: true, enumerable: true, configurable: true } }
‚úÖ Object.defineProperty() & Object.defineProperties()
const obj = {};

// Define single property
Object.defineProperty(obj, "name", {
    value: "John",
    writable: false,
    enumerable: true,
    configurable: true
});

// Define multiple properties
Object.defineProperties(obj, {
    age: {
        value: 30,
        writable: true,
        enumerable: true
    },
    email: {
        value: "john@email.com",
        writable: true
    }
});

console.log(obj); // { name: "John", age: 30, email: "john@email.com" }
obj.name = "Bob"; // Fails silently (writable: false)
‚úÖ Object.is() - Strict Equality Check
// Object.is() vs ===
console.log(Object.is(5, 5)); // true
console.log(Object.is("5", 5)); // false
console.log(Object.is(NaN, NaN)); // true ‚úì (Diff from ===)
console.log(NaN === NaN); // false
console.log(Object.is(+0, -0)); // false ‚úì (Diff from ===)
console.log(+0 === -0); // true
‚úÖ Object.hasOwn() vs hasOwnProperty()
const obj = { name: "John" };
const proto = { age: 30 };
Object.setPrototypeOf(obj, proto);

// Object.hasOwn() - Recommended
console.log(Object.hasOwn(obj, "name")); // true
console.log(Object.hasOwn(obj, "age")); // false
console.log(Object.hasOwn(obj, "toString")); // false

// hasOwnProperty() - Legacy
console.log(obj.hasOwnProperty("name")); // true
console.log(obj.hasOwnProperty("age")); // false

5. Object Cloning & Merging

‚úÖ Shallow Copy - Spread Operator
const original = { name: "John", age: 30 };

// Shallow copy using spread operator
const copy = { ...original };
copy.name = "Bob";

console.log(original); // { name: "John", age: 30 }
console.log(copy); // { name: "Bob", age: 30 }

// ‚ö†Ô∏è Nested objects are still referenced
const user = { name: "John", address: { city: "NYC" } };
const shallowCopy = { ...user };
shallowCopy.address.city = "LA";
console.log(user.address.city); // "LA" (Original affected!)
‚úÖ Shallow Copy - Object.assign()
const original = { name: "John", age: 30 };

// Shallow copy
const copy = Object.assign({}, original);
copy.name = "Bob";

console.log(original); // { name: "John", age: 30 }
console.log(copy); // { name: "Bob", age: 30 }
‚úÖ Deep Copy - JSON Method
const original = { name: "John", address: { city: "NYC" } };

// Deep copy (works for serializable objects)
const deepCopy = JSON.parse(JSON.stringify(original));
deepCopy.address.city = "LA";

console.log(original.address.city); // "NYC" (Original NOT affected)
console.log(deepCopy.address.city); // "LA"

// ‚ö†Ô∏è Issues: Functions, Symbols, undefined are lost
const obj = { fn: () => {}, val: undefined };
const copy = JSON.parse(JSON.stringify(obj));
console.log(copy); // {} (Function and undefined lost)
‚úÖ Deep Copy - Recursive Function
function deepClone(obj) {
    if (obj === null || typeof obj !== "object") return obj;
    if (obj instanceof Date) return new Date(obj);
    if (obj instanceof Array) return obj.map(item => deepClone(item));
    
    const cloned = {};
    for (let key in obj) {
        if (obj.hasOwnProperty(key)) {
            cloned[key] = deepClone(obj[key]);
        }
    }
    return cloned;
}

const original = { name: "John", address: { city: "NYC" } };
const copy = deepClone(original);
copy.address.city = "LA";

console.log(original.address.city); // "NYC"
‚úÖ Merging Objects
const obj1 = { a: 1, b: 2 };
const obj2 = { b: 3, c: 4 };
const obj3 = { d: 5 };

// Using Object.assign()
const merged1 = Object.assign({}, obj1, obj2, obj3);
console.log(merged1); // { a: 1, b: 3, c: 4, d: 5 }

// Using spread operator
const merged2 = { ...obj1, ...obj2, ...obj3 };
console.log(merged2); // { a: 1, b: 3, c: 4, d: 5 }

// Note: Later values override earlier ones
const merged3 = { ...obj1, ...obj2 }; // obj2.b overrides obj1.b

6. Getters & Setters

‚úÖ Getter & Setter Syntax
const user = {
    _name: "",
    
    get name() {
        return this._name;
    },
    
    set name(value) {
        if (typeof value !== "string") {
            console.log("Name must be a string");
            return;
        }
        this._name = value;
    }
};

user.name = "John";
console.log(user.name); // "John"

user.name = 123; // Validation fails
console.log(user.name); // "John" (Not changed)
‚úÖ Getter & Setter in Class
class Circle {
    constructor(radius) {
        this._radius = radius;
    }
    
    get radius() {
        return this._radius;
    }
    
    set radius(value) {
        if (value <= 0) {
            console.log("Radius must be positive");
            return;
        }
        this._radius = value;
    }
    
    get area() {
        return Math.PI * this._radius ** 2;
    }
}

const circle = new Circle(5);
console.log(circle.radius); // 5
console.log(circle.area); // 78.54

circle.radius = 10;
console.log(circle.area); // 314.16

7. Object Methods

‚úÖ Adding Methods to Objects
const user = {
    name: "John",
    age: 30,
    
    // Method 1: Function Declaration
    greet() {
        return "Hello, I'm " + this.name;
    },
    
    // Method 2: Function Expression
    sayAge: function() {
        return "I'm " + this.age + " years old";
    }
};

console.log(user.greet()); // "Hello, I'm John"
console.log(user.sayAge()); // "I'm 30 years old"
‚úÖ Method Chaining
const calculator = {
    value: 0,
    
    add(n) {
        this.value += n;
        return this;
    },
    
    subtract(n) {
        this.value -= n;
        return this;
    },
    
    multiply(n) {
        this.value *= n;
        return this;
    },
    
    getResult() {
        return this.value;
    }
};

const result = calculator.add(10).multiply(2).subtract(5).getResult();
console.log(result); // 15
‚úÖ call(), apply(), bind()
function greet(greeting, punctuation) {
    return greeting + ", " + this.name + punctuation;
}

const person1 = { name: "John" };
const person2 = { name: "Alice" };

// call() - pass arguments individually
console.log(greet.call(person1, "Hello", "!")); 
// "Hello, John!"

// apply() - pass arguments as array
console.log(greet.apply(person2, ["Hi", "?"])); 
// "Hi, Alice?"

// bind() - creates new function with bound context
const greetJohn = greet.bind(person1);
console.log(greetJohn("Hey", "~")); 
// "Hey, John~"

9. Symbol Properties

‚úÖ Using Symbols as Keys
// Create unique symbols
const id = Symbol("id");
const email = Symbol("email");

const user = {
    name: "John",
    [id]: 12345,
    [email]: "john@email.com"
};

// Access symbol properties
console.log(user[id]); // 12345

// Symbols don't appear in Object.keys()
console.log(Object.keys(user)); // ["name"]

// Get symbol properties

// [Symbol(id), Symbol(email)]
‚úÖ Symbol.for() & Symbol.keyFor()
// Symbol.for() creates global symbols
const globalSym1 = Symbol.for("app.id");
const globalSym2 = Symbol.for("app.id");

console.log(globalSym1 === globalSym2); // true (Same symbol)

// Symbol.keyFor() gets the key
console.log(Symbol.keyFor(globalSym1)); // "app.id"

// Local symbols vs Global symbols
const localSym = Symbol("id");
console.log(Symbol.keyFor(localSym)); // undefined

10. Advanced Patterns

‚úÖ Private Fields
class BankAccount {
    #balance = 0; // Private field
    
    constructor(initialBalance) {
        this.#balance = initialBalance;
    }
    
    deposit(amount) {
        this.#balance += amount;
    }
    
    getBalance() {
        return this.#balance;
    }
}

const account = new BankAccount(1000);
console.log(account.getBalance()); // 1000
account.deposit(500);
console.log(account.getBalance()); // 1500
console.log(account.#balance); // Error: Private field
‚úÖ Singleton Pattern
const Singleton = (() => {
    let instance;
    
    return {
        getInstance() {
            if (!instance) {
                instance = {
                    name: "Singleton",
                    value: Math.random()
                };
            }
            return instance;
        }
    };
})();

const obj1 = Singleton.getInstance();
const obj2 = Singleton.getInstance();
console.log(obj1 === obj2); // true (Same instance)
‚úÖ Factory Pattern
function createUser(name, role) {
    return {
        name,
        role,
        getPermissions() {
            if (role === "admin") return ["read", "write", "delete"];
            if (role === "user") return ["read", "write"];
            return ["read"];
        }
    };
}

const admin = createUser("John", "admin");
const user = createUser("Alice", "user");

console.log(admin.getPermissions());
// ["read", "write", "delete"]
console.log(user.getPermissions());
// ["read", "write"]
‚úÖ Module Pattern
const UserModule = (() => {
    let users = []; // Private
    
    return {
        addUser(user) {
            users.push(user);
        },
        
        getUsers() {
            return [...users]; // Return copy
        },
        
        removeUser(id) {
            users = users.filter(u => u.id !== id);
        }
    };
})();

UserModule.addUser({ id: 1, name: "John" });
UserModule.addUser({ id: 2, name: "Alice" });
console.log(UserModule.getUsers());
// [{ id: 1, name: "John" }, { id: 2, name: "Alice" }]
